          DB.getFamilies()

              const item = document.createElement("div");
              item.className =
                "list-item" + (selectedFamId === fam.id ? " active" : "");

                renderLeftPanel();
                renderFamilyEditor(fam.id);

          DB.getEtablissements()
            .filter((e) => e.nom.toLowerCase().includes(search))

      function renderClassesContainer(famId) {
        const fam = DB.getFamily(famId);
        const container = document.getElementById("classesContainer");
        if (!container || !fam) return;
        container.innerHTML = "";
        fam.classes.forEach((cls, ci) => {
          const block = document.createElement("div");
          block.className = "class-block";
          block.innerHTML = `
      <div class="class-block-header">
        <div class="class-color-dot" style="background:${cls.couleur}"></div>
        <input class="class-name-input" value="${cls.nom}" placeholder="Nom de la classe" onchange="updateClassProp('${famId}',${ci},'nom',this.value)">
        <select class="color-select" onchange="updateClassProp('${famId}',${ci},'couleur',this.value)">
          ${["#6c63ff", "#48bb78", "#9f7aea", "#f6ad55", "#fc5c5c", "#63b3ed", "#4fd1c5", "#ed8936"].map((c) => `<option value="${c}" ${cls.couleur === c ? "selected" : ""}>${c}</option>`).join("")}
        </select>
        <button class="del-cls-btn" onclick="deleteClassBlock('${famId}',${ci})">
          <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="3 6 5 6 21 6"/><path d="M19 6l-1 14H6L5 6"/></svg>
        </button>
      </div>
      <div class="vars-area" id="vars_${ci}">
        ${cls.vars
          .map(
            (v, vi) => `
          <div class="var-pill" style="background:${cls.couleur}15;border-color:${cls.couleur}40;color:${cls.couleur}">
            <span class="var-pill-tech">{{${v.tech}}}</span>
            <span class="var-pill-label">→ ${v.label}</span>
            <button class="var-pill-del" onclick="deleteVar('${famId}',${ci},${vi})">×</button>
          </div>`,
          )
          .join("")}
      </div>
      <div class="var-add-row">
        <input class="var-input-sm" id="vt_${ci}" placeholder="nom_technique">
        <input class="var-input-sm label" id="vl_${ci}" placeholder="Nom lisible pour l'admin">
        <button class="btn sm primary" onclick="addVar('${famId}',${ci})">

      function updateClassProp(famId, ci, prop, val) {
        const fam = DB.getFamily(famId);
        fam.classes[ci][prop] = val;
        DB.saveFamily(fam);
        if (prop === "couleur") renderClassesContainer(famId);
      }

      function addClassBlock(famId) {
        const fam = DB.getFamily(famId);
        fam.classes.push({
          nom: "Nouvelle classe",
          couleur: "#6c63ff",
          vars: [],
        });
        DB.saveFamily(fam);
        renderClassesContainer(famId);
      }

      function deleteClassBlock(famId, ci) {
        const fam = DB.getFamily(famId);
        fam.classes.splice(ci, 1);
        DB.saveFamily(fam);
        renderClassesContainer(famId);
      }

      function addVar(famId, ci) {
        const tech = document
          .getElementById(`vt_${ci}`)
          ?.value.trim()
          .replace(/\s/g, "_");
        const label = document.getElementById(`vl_${ci}`)?.value.trim();
        if (!tech || !label) {
          toast("Remplissez les deux champs", "error");
          return;
        }
        const fam = DB.getFamily(famId);
        fam.classes[ci].vars.push({ tech, label });
        DB.saveFamily(fam);
        renderClassesContainer(famId);
        document.getElementById(`vt_${ci}`).value = "";
        document.getElementById(`vl_${ci}`).value = "";
      }

      function deleteVar(famId, ci, vi) {
        const fam = DB.getFamily(famId);
        fam.classes[ci].vars.splice(vi, 1);
        DB.saveFamily(fam);
        renderClassesContainer(famId);
      }

      function fmt(format, value) {
        var q = aqQ || qB;
        if (!q) return;
        var r = q.getSelection(true);
        if (!r) return;
        if (value !== undefined) {
          var curr = q.getFormat(r);
          q.format(format, curr[format] === value ? false : value);
        } else {
          var curr2 = q.getFormat(r);
          q.format(format, !curr2[format]);
        }
        updateRibbonState();
      }

      function align(dir) {
        var q = aqQ || qB;
        if (!q) return;
        q.format("align", dir === "left" ? false : dir);
        updateRibbonState();
      }

      function indent(delta) {
        var q = aqQ || qB;
        if (!q) return;
        var r = q.getSelection(true);
        if (!r) return;
        var curr = q.getFormat(r);
        var lv = (curr.indent || 0) + delta;
        q.format("indent", Math.max(0, Math.min(8, lv)));
      }

      function buildTableHTML(rows, cols, style) {
        var hdrBg = "#ebf0ff",
          hdrColor = "#1a56db",
          border = "1px solid #ddd";
        var evenBg = "transparent",
          oddBg = "#f8f9fc";
        if (style === "minimal") {
          hdrBg = "#f7f7f4";
          hdrColor = "#333";
          border = "1px solid #eee";
        }
        if (style === "noborder") {
          border = "none";
          hdrBg = "transparent";
          hdrColor = "#555";
        }
        var html =
          '<table style="border-collapse:collapse;width:100%;margin:8px 0">';
        for (var r = 0; r < rows; r++) {
          html += "<tr>";
          for (var c = 0; c < cols; c++) {
            if (
              r === 0 &&
              (style === "default" || style === "header" || style === "striped")
            ) {
              html +=
                '<th style="padding:8px 10px;background:' +
                hdrBg +
                ";color:" +
                hdrColor +
                ";font-weight:600;border:" +
                border +
                ';text-align:left">En-tête ' +
                (c + 1) +
                "</th>";
            } else {
              var bg =
                style === "striped" && r % 2 === 0 ? oddBg : "transparent";
              html +=
                '<td style="padding:8px 10px;border:' +
                border +
                ";background:" +
                bg +
                '"></td>';
            }
          }
          html += "</tr>";
        }
        html += "</table>";
        return html;
      }

      function openImgModal() {
        openModal("modalImage");

      function previewImgFile(input) {
        var file = input.files[0];
        if (!file) return;
        var reader = new FileReader();
        reader.onload = function (e) {
          document.getElementById("img-preview").src = e.target.result;
          document.getElementById("img-preview-area").style.display = "flex";
        };
        reader.readAsDataURL(file);
      }

      function previewImgUrl(url) {
        if (!url) return;
        document.getElementById("img-preview").src = url;
        document.getElementById("img-preview-area").style.display = "flex";
      }

      function doInsertImage() {
        var q = aqQ || qB;
        if (!q) return;
        var fileInput = document.getElementById("img-file");
        var url = document.getElementById("img-url").value;
        var w = document.getElementById("img-w").value;
        var alignVal = document.getElementById("img-align").value;
        var caption = document.getElementById("img-caption").value;

        var wStyle = w
          ? "width:" + w + "px;max-width:100%;"
          : "max-width:100%;";
        var containerStyle = "margin:8px 0;";
        if (alignVal === "center") containerStyle += "text-align:center;";
        else if (alignVal === "right") containerStyle += "text-align:right;";
        else if (alignVal === "left") containerStyle += "text-align:left;";

        function doInsert(src) {
          var r = q.getSelection(true);
          var idx = r ? r.index : q.getLength() - 1;
          var capHtml = caption
            ? '<p style="font-size:11px;color:#888;text-align:center;margin-top:2px"><em>' +
              escHtml(caption) +
              "</em></p>"
            : "";
          q.clipboard.dangerouslyPasteHTML(
            idx,
            '<div style="' +
              containerStyle +
              '"><img src="' +
              src +
              '" style="' +
              wStyle +
              'border-radius:4px" alt="image"></div>' +
              capHtml +
              "<p><br></p>",
          );
          closeModal("modalImage");
          toast("Image insérée");
          document.getElementById("img-file").value = "";
          document.getElementById("img-url").value = "";
          document.getElementById("img-preview-area").style.display = "none";
          document.getElementById("img-w").value = "";
          document.getElementById("img-caption").value = "";
        }

        if (fileInput.files[0]) {
          var reader = new FileReader();
          reader.onload = function (e) {
            doInsert(e.target.result);
          };
          reader.readAsDataURL(fileInput.files[0]);
        } else if (url) {
          doInsert(url);
        } else {
          toast("Sélectionnez une image ou entrez une URL", "error");
        }
      }

      function openLinkModal() {
        var q = aqQ || qB;
        if (q) {
          var r = q.getSelection();
          if (r && r.length > 0)
            document.getElementById("link-text").value = q.getText(
              r.index,
              r.length,
            );
        }
        openModal("modalLink");
      }

      function doInsertLink() {
        var q = aqQ || qB;
        if (!q) return;
        var text = document.getElementById("link-text").value;
        var url = document.getElementById("link-url").value;
        if (!url) {
          toast("URL requise", "error");
          return;
        }
        var r = q.getSelection(true);
        var idx = r ? r.index : q.getLength() - 1;
        if (text) q.insertText(idx, text, { link: url, color: "#1a56db" });
        else q.format("link", url);
        closeModal("modalLink");
        document.getElementById("link-text").value = "";
        document.getElementById("link-url").value = "";
        toast("Lien inséré");
      }

      // ===================================================
      //  SAVE / DELETE
      // ===================================================
      function saveTemplate() {
        if (!curTplId || !qB) return;
        var tpl = DB.getTemplate(curTplId);
        var nameInput = document.getElementById("tnameInput");
        if (nameInput) tpl.nom = nameInput.value || tpl.nom;
        tpl.header = qH ? qH.root.innerHTML : "";
        tpl.body = qB.root.innerHTML;
        tpl.footer = qF ? qF.root.innerHTML : "";
        tpl.updatedAt = new Date().toISOString();
        DB.saveTemplate(tpl);
        renderTplList(curFamId);
        var st = document.getElementById("saveSt");
        if (st) {
          st.textContent = "Enregistré ✓";
          st.className = "save-st";
        }
        var dot = document.getElementById("sbDot");
        if (dot) dot.className = "sb-dot ok";
        var txt = document.getElementById("sbText");
        if (txt) txt.textContent = "Enregistré";
        toast("Template enregistré", "success");
      }

      function confirmDelTpl(id) {
        pendingDelId = id;
        document.getElementById("delTplBtn").onclick = function () {
          DB.deleteTemplate(pendingDelId);
          if (curTplId === pendingDelId) {
            curTplId = null;
            clearEditor();
          }
          renderTplList(curFamId);
          closeModal("modalDelTpl");
          toast("Template supprimé");
        };
        openModal("modalDelTpl");
      }

      function openPreviewModal() {
        if (!curTplId || !qB) return;
        var sel = document.getElementById("prevPerson");
        sel.innerHTML = "";
        DB.getPersonnel(ETAB_ID).forEach(function (p) {
          var o = document.createElement("option");
          o.value = p.id;
          o.textContent = p.nom_prenom + " — " + p.poste;
          sel.appendChild(o);
        });
        refreshPrev();
        openModal("modalPreview");
      }

      function refreshPrev() {
        var pId = document.getElementById("prevPerson").value;
        var person = DB.getPerson(pId);
        if (!person || !qB) return;
        var raw =
          (qH ? qH.root.innerHTML : "") +
          '<hr style="border:none;border-top:1px solid #eee;margin:10px 0">' +
          qB.root.innerHTML +
          '<hr style="border:none;border-top:1px solid #eee;margin:10px 0">' +
          (qF ? qF.root.innerHTML : "");
        document.getElementById("prevContent").innerHTML = resolveVars(
          raw,
          person,
        );
      }

      // ===================================================
      //  RIBBON STATE
      // ===================================================
      function updateRibbonState() {
        var q = aqQ || qB;
        if (!q) return;
        var r = q.getSelection();
        if (!r) return;
        var f = q.getFormat(r);
        ["bold", "italic", "underline", "strike"].forEach(function (k) {
          var btn = document.getElementById("rb-" + k);
          if (btn) btn.classList.toggle("on", !!f[k]);
        });
        var al = f.align || "left";
        var alMap = { al: "left", ac: "center", ar: "right", aj: "justify" };
        Object.keys(alMap).forEach(function (k) {
          var btn = document.getElementById("rb-" + k);
          if (btn)
            btn.classList.toggle(
              "on",
              al === alMap[k] || (k === "al" && !f.align),
            );
        });
        var bul = document.getElementById("rb-bul");
        if (bul) bul.classList.toggle("on", f.list === "bullet");
        var ord = document.getElementById("rb-ord");
        if (ord) ord.classList.toggle("on", f.list === "ordered");
      }

      function onTxtChange() {
        var st = document.getElementById("saveSt");
        if (st) {
          st.textContent = "Modifié";
          st.className = "save-st mod";
        }
        var dot = document.getElementById("sbDot");
        if (dot) dot.className = "sb-dot mod";
        var txt = document.getElementById("sbText");
        if (txt) txt.textContent = "Modifié";
        var bodyQ = document.getElementById("eq-body");
        if (bodyQ) {
          var text =
            (bodyQ.querySelector(".ql-editor") || {}).textContent || "";
          var words = text.trim() ? text.trim().split(/\s+/).length : 0;
          var sw = document.getElementById("sbWords");
          if (sw) sw.textContent = words + " mots";
          var sc = document.getElementById("sbChars");
          if (sc) sc.textContent = text.length + " car.";
        }
      }

      // ===================================================
      //  UNDO / REDO

      function undoAction() {
        var q = aqQ || qB;
        if (q && q.history) q.history.undo();
      }
      function redoAction() {
        var q = aqQ || qB;
        if (q && q.history) q.history.redo();
      }

      // ===================================================

      function showCtxMenu(e) {
        var menu = document.getElementById("ctxMenu");
        menu.style.left = Math.min(e.clientX, window.innerWidth - 190) + "px";
        menu.style.top = Math.min(e.clientY, window.innerHeight - 280) + "px";
        menu.classList.add("on");
      }
      function closeCtx() {
        document.getElementById("ctxMenu").classList.remove("on");
      }
      function ctxPaste() {
        navigator.clipboard.readText().then(function (t) {
          var q = aqQ || qB;
          if (!q) return;
          var r = q.getSelection(true);
          var idx = r ? r.index : 0;
          q.insertText(idx, t);
        });
        closeCtx();
      }
      function delSel() {
        var q = aqQ || qB;
        if (!q) return;
        var r = q.getSelection();
        if (r) q.deleteText(r.index, r.length);
      }

      // ===================================================
      //  MODAL UTILS
      // ===================================================
      function openModal(id) {
        var el = document.getElementById(id);
        if (el) el.classList.add("open");
      }
      function closeModal(id) {
        var el = document.getElementById(id);
        if (el) el.classList.remove("open");
      }

      // ===================================================
      //  KEYBOARD SHORTCUTS
      // ===================================================
      document.addEventListener("keydown", function (e) {
        if ((e.ctrlKey || e.metaKey) && e.key === "s") {
          e.preventDefault();
          saveTemplate();
        }
        if ((e.ctrlKey || e.metaKey) && e.key === "z" && !e.shiftKey) {
          e.preventDefault();
          undoAction();
        }
        if (
          (e.ctrlKey || e.metaKey) &&
          (e.key === "y" || (e.key === "z" && e.shiftKey))
        ) {
          e.preventDefault();
          redoAction();
        }
        if ((e.ctrlKey || e.metaKey) && e.key === "p") {
          e.preventDefault();
          window.print();
        }
        if (e.key === "Escape") {
          closeCtx();
          closeAllCpops();
          document
            .querySelectorAll(".modal-overlay.open")
            .forEach(function (m) {
              m.classList.remove("open");
            });
        }
      });
      document.addEventListener("click", function (e) {
        closeCtx();
        if (!e.target.closest(".rcolor")) closeAllCpops();
        if (e.target.classList.contains("modal-overlay"))
          e.target.classList.remove("open");
      });
